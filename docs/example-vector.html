Vector Operations Example

GPU-accelerated vector multiply demonstration

Overview

This example demonstrates basic vector operations using Parallax unified memory.

Complete Code

C++ vector_multiply.cpp 
#include <parallax/runtime.h>
#include <iostream>
#include <vector>

int main() {
    const size_t N = 1'000'000;
    const float multiplier = 2.0f;
    
    // Allocate unified memory
    float* input = (float*)parallax_umalloc(N * sizeof(float), 0);
    float* output = (float*)parallax_umalloc(N * sizeof(float), 0);
    
    // Initialize input
    for (size_t i = 0; i < N; i++) {
        input[i] = static_cast<float>(i);
    }
    
    // Sync to GPU
    parallax_sync(input, 0);
    
    // GPU computation would happen here
    // For now, simulate with CPU
    for (size_t i = 0; i < N; i++) {
        output[i] = input[i] * multiplier;
    }
    
    // Sync back
    parallax_sync(output, 1);
    
    // Verify
    std::cout << "First: " << output[0] << std::endl;
    std::cout << "Last: " << output[N-1] << std::endl;
    
    // Cleanup
    parallax_ufree(input);
    parallax_ufree(output);
    
    return 0;
}

Build and Run

terminal 
g++ -std=c++20 vector_multiply.cpp -lparallax-runtime -o vector_multiply
./vector_multiply
