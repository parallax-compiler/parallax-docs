parallax_demo.cpp 
// Your existing C++ code - unchanged
std::vector<float> data(1'000'000, 1.0f);

// Just use std::execution::par - Parallax handles the rest
std::for_each(std::execution::par, 
              data.begin(), data.end(),
              [](float& x) { x *= 2.0f; });

// Memory coherence is automatic - no sync needed!
// Works on AMD, NVIDIA, Intel - anything with Vulkan
